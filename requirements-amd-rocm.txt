# AMD ROCm 用の追加パッケージ
#
# 事前準備:
#   1. ROCmのインストール（詳細はREADME.mdの「ROCm環境セットアップ」を参照）
#   2. PyTorch (ROCm版) のインストール:
#      pip install torch torchvision --index-url https://download.pytorch.org/whl/rocm5.7
#   3. 環境変数の設定（~/.bashrc に追加）:
#      export PATH=/opt/rocm/bin:$PATH
#      export LD_LIBRARY_PATH=/opt/rocm/lib:$LD_LIBRARY_PATH
#      export HIP_PATH=/opt/rocm
#      source ~/.bashrc
#
# インストール:
#   pip install paddlepaddle-gpu
#   uv pip install paddlepaddle-gpu
#
# 対応GPU: AMD AI MAX 395+, Radeon RX 6000/7000, Radeon Pro, Instinctシリーズ
# 対応OS: Ubuntu 20.04/22.04, CentOS/RHEL 7.9/8.5, Fedora 34+（Linuxのみ）
#
# 注意: ROCmはmacOS / Windows では動作しません
